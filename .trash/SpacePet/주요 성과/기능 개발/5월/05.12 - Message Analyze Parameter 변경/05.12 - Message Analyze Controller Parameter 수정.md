### MessageService

```kotlin
/**  
 * @author 신건우  
 * @param id : 메시지 아이디  
 * @param title : 제목  
 * @param body : 내용  
 * @param type : 메시지 타입  
 * @param start : 검색 시작 날짜  
 * @param end : 검색 종료 날짜  
 * @param createdAt : 메시지 생성 일자  
 * @desc Message Analyze 전체 조건 검색 Admin 기능 (페이지네이션)  
 */fun getAllMessageAnalyze(  
    id: String?,  
    title: String?,  
    body: String?,  
    type: String?,  
    start: LocalDate?,  
    end: LocalDate?,  
    createdAt: LocalDateTime?,  
): List<MessageAnalyzeDTO.Detail>?  {  
  
    val list = try {  
        messageAnalyzeRepository.findByCondition(id, title, body, type, start, end, createdAt)  
    } catch (e: Exception) {  
        log.error("Message Analyze 전체 조회 실패 - $e")  
        throw CommonException("Message Analyze 전체 조회 실패", HttpStatus.BAD_REQUEST)  
    }  
  
    log.info("Message Analyze 전체 조건 검색 성공 - 메시지 ID : $id")  
  
    return list?.stream()?.map { MessageAnalyzeDTO.Detail.fromEntity(it) }?.toList()  
}
```

```kotlin
/**  
 * @author 신건우  
 * @desc Create Query Factory  
 */override fun findByCondition(  
    id: String?,  
    title: String?,  
    body: String?,  
    type: String?,  
    start: LocalDate?,  
    end: LocalDate?,  
    createdAt: LocalDateTime?,  
): MutableList<MessageAnalyze>? {  
  
    val paging = PageRequest.of(0, 30)  
  
    val builder = searchCondition(id, title, body, type, start, end, createdAt)  
  
    return queryFactory  
        .selectFrom(messageAnalyze)  
        .where(builder)  
        .orderBy(messageAnalyze.createdAt.desc())  
        .offset(paging.offset)  
        .limit(30.toLong())  
        .fetch()  
}  
  
/**  
 * @author 신건우  
 * @desc QueryDsl - Message Analyze 조건 검색 BooleanBuilder */fun searchCondition(  
    id: String?,  
    title: String?,  
    body: String?,  
    type: String?,  
    start: LocalDate?,  
    end: LocalDate?,  
    createdAt: LocalDateTime?,  
): BooleanBuilder? {  
  
    val builder = BooleanBuilder()  
    val startDate = start?.let { LocalDateTime.of(it.year, it.month, it.dayOfMonth, 0, 0 ,0) }  
    val endDate = end?.let { LocalDateTime.of(it.year, it.month, it.dayOfMonth, 23 ,59, 59)}  
  
    fun eqId(id: String): BooleanExpression? {  
        return if (id.isEmpty()) null else messageAnalyze.id.contains(id)  
    }  
  
    fun eqTitle(title: String): BooleanExpression? {  
        return if (title.isEmpty()) null else messageAnalyze.title.contains(title)  
    }  
  
    fun eqBody(body: String): BooleanExpression? {  
        return if (body.isEmpty()) null else messageAnalyze.body.contains(body)  
    }  
  
    fun eqType(type: String): BooleanExpression? {  
        return if (type.isEmpty()) null else messageAnalyze.type.contains(type)  
    }  
  
    fun eqCreatedAt(createdAt: LocalDateTime): BooleanExpression? {  
        return messageAnalyze.createdAt.eq(createdAt)  
    }  
  
    builder.or(id?.let { eqId(it) })  
  
    builder.or(title?.let { eqTitle(it) })  
  
    builder.or(body?.let { eqBody(it) })  
  
    builder.or(type?.let { eqType(it) })  
  
    builder.or(createdAt?.let { eqCreatedAt(it) })  
  
    if (start != null && end != null) {  
        builder.and(messageAnalyze.createdAt.between(startDate, endDate))  
    } else if (start != null) {  
        builder.and(messageAnalyze.createdAt.goe(startDate))  
    } else if (end != null) {  
        builder.and(messageAnalyze.createdAt.loe(endDate))  
    }  
  
    return builder  
}
```

```kotlin
@Transactional  
fun postMessageAnalyze(  
    id: String?,  
    title: String?,  
    body: String?,  
    type: String?  
): MessageAnalyzeDTO.Detail {  
  
    val message = MessageAnalyze(  
        id = UUID.randomUUID().toString().substring(0, 8),  
        title = title!!,  
        body = body!!,  
        type = type!!,  
        data = "",  
        send = 0,  
        reach = 0,  
        click = 0,  
        createdAt = LocalDateTime.now()  
    )  
  
    messageAnalyzeRepository.save(message)  
    messageAnalyzeRepository.flush()  
  
    log.info("테스트 메시지 생성 완료")  
    log.info("Create Test MessageAnalyze Entity Response")  
  
    return MessageAnalyzeDTO.Detail.fromEntity(message)  
}
```
