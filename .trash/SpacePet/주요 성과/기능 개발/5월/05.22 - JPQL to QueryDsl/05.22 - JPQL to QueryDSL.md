### CustomUserRepositoryImpl

```kotlin
override fun findByEmail(email: String): User? {
    return from(user)
        .where(user.email.eq(email))
        .fetchOne()
}

override fun findByDummyName(dummyName: String): User {
    return from(user)
        .where(user.name.eq(dummyName))
        .fetchOne()
}

override fun findByIds(users: List<String>): List<User> {
    return from(user)
        .where(
            user.deleted.eq(false)
            .and(user.id.notIn(users))
        )
        .orderBy(user.id.desc())
        .fetch()
}

override fun findByName(name: String): List<User> {  
    return from(user)  
        .where(  
            user.deleted.eq(false)  
                .and(user.name.eq(name))  
        )  
        .fetch()  
}

// totalUserCount = total count u  
// dateUserCount = user deleted=false,grade=NOT IN '탈퇴회원', createdAt between startCD/endCD -> count u  
// mainMember = grade=정회원, deleted=false -> count u  
// subMember = grade=부회원,deleted false  
// user = deleted=false, grade=NOT IN '탈퇴회원' -> grade=준회원, deleted=false  
override fun findUserCount(startCurrentDate: LocalDateTime, endCurrentDate: LocalDateTime): Any {  
    val totalUserCount = from(user)  
        .where(user.deleted.isFalse, user.grade.notIn("탈퇴회원"))  
        .fetchCount()  
  
    val dateUserCount = from(user)  
        .leftJoin(user.userDate)  
        .where(  
            user.userDate.createdAt.between(startCurrentDate, endCurrentDate),  
            user.deleted.isFalse,  
            user.grade.notIn("탈퇴회원")  
        )  
        .fetchCount()  
  
    val mainMemberCount = from(user)  
        .where(user.grade.eq("정회원"), user.deleted.isFalse)  
        .fetchCount()  
  
    val subMemberCount = from(user)  
        .where(user.grade.eq("부회원"), user.deleted.isFalse)  
        .fetchCount()  
  
    val juniorMemberCount = from(user)  
        .where(user.grade.eq("준회원"), user.deleted.isFalse)  
        .fetchCount()  
  
    return mapOf(  
        "totalUserCount" to totalUserCount,  
        "dateUserCount" to dateUserCount,  
        "mainMember" to mainMemberCount,  
        "subMember" to subMemberCount,  
        "user" to juniorMemberCount  
    )  
}

override fun findUserRetention(): Any {  
    val userDate = user.userDate  
    val token = user.userToken  
  
    val retention = queryFactory  
        .select(  
            userDate.createdAt.count().`as`("firstTotal"),  
            userDate.createdAt.count().`as`("firstUser"),  
            userDate.createdAt.count().`as`("secondTotal"),  
            userDate.createdAt.count().`as`("secondUser"),  
            userDate.createdAt.count().`as`("thirdTotal"),  
            userDate.createdAt.count().`as`("thirdUser"),  
            userDate.createdAt.count().`as`("fourthTotal"),  
            userDate.createdAt.count().`as`("fourthUser"),  
            userDate.createdAt.count().`as`("fifthTotal"),  
            userDate.createdAt.count().`as`("fifthUser"),  
            userDate.createdAt.count().`as`("sixthTotal"),  
            userDate.createdAt.count().`as`("sixthUser"),  
            userDate.createdAt.count().`as`("seventhTotal"),  
            userDate.createdAt.count().`as`("seventhUser")  
        )  
        .from(user)  
        .join(user.userDate, userDate)  
        .leftJoin(user.userToken, token)  
        .on(token.tokenCreatedAt.goe(userDate.createdAt))  
        .where(  
            userDate.createdAt.loe(LocalDateTime.now().minusDays(1)),  
            userDate.createdAt.loe(LocalDateTime.now().minusDays(2)),  
            userDate.createdAt.loe(LocalDateTime.now().minusDays(3)),  
            userDate.createdAt.loe(LocalDateTime.now().minusDays(4)),  
            userDate.createdAt.loe(LocalDateTime.now().minusDays(5)),  
            userDate.createdAt.loe(LocalDateTime.now().minusDays(6)),  
            userDate.createdAt.loe(LocalDateTime.now().minusDays(7))  
        )  
        .fetchOne()  
  
    return retention  
}

override fun updateUsersGrade(grade: String, ids: List<String>) {  
    update(user)  
        .set(user.grade, grade)  
        .where(user.id.`in`(ids))  
        .execute()  
}

override fun updateUserMainGrade(grade: String, ids: List<String>) {  
    update(user)  
        .set(user.grade, grade)  
        .where(user.grade.eq("부회원"), user.id.`in`(ids))  
        .execute()  
}  
  
override fun findByDeletedUserOrderByProfileAtDesc(): List<User> {  
    return from(user)  
        .where(user.deleted.eq(true))  
        .orderBy(user.userDate.profileAt.desc().nullsLast())  
        .fetch()  
}  
  
override fun findByGradeOrderByProfileAtDesc(grade: String): List<User> {  
    return from(user)  
        .where(user.deleted.eq(false), user.grade.eq((grade)))  
        .orderBy(user.userDate.profileAt.desc().nullsLast())  
        .fetch()  
}  
  
override fun findByRequestReAuthTrueOrderByReAuthAtAsc(): List<User> {  
    return from(user)  
        .where(user.userDate.requestReAuth.eq(true), user.grade.eq("부회원"), user.deleted.eq(false))  
        .orderBy(user.userDate.reAuthAt.asc())  
        .fetch()  
}
```
