## 요구사항

현재 분리된 앨범 사진 업로드와 투데이펫 업로드 기능을 앨범 사진 업로드 서비스에 합칠 것

- inputStream은 한번만 사용 가능
- NCPS3 서비스로 이미지 전송기능 이관
  범용성 있게 짜기 (앨범 / 보드) (폴더 이름 기준으로 여러개도 넣을 수 있게)
  반환값은 입력값과 같게 
- 이미지 업로드 중간에 실패시 어떻게 해야 할지,
  한장이라도 들어간다면 NCP 디렉토리 삭제하기.

```kotlin
@Transactional
    fun postAlbum (files : List<MultipartFile>, body : AlbumPostDTO, uid : String) : AlbumDTO {

        var anUser : User = userRecycleFn.checkActiveUser(uid)

        var anAlbum : Album = try {
            postAlbumEntity(body, uid)
        } catch (e : Exception) {
            log.error("ALBUM-003 : 앨범 생성 실패 - [ 유저명 : $uid, $e ]")
            throw CommonException("ALBUM-003", HttpStatus.BAD_REQUEST)
        }

        // 투데이펫 게시글도 같이 올릴 경우
        if (body.todayPet == true) {

            // 투데이펫 게시글일 경우 2개 리스트
            var albumImages = mutableListOf<String>()
            var boardImages = mutableListOf<String>()

            // 보드 생성
            val aBoard: Board = try {
                Board(user = anUser).apply {
                    kind = "일상"
                    requestContent = body.content
                    createdAt = LocalDateTime.now()
                    pushUsers = mutableListOf(uid)
                }
            } catch (e: Exception) {
                log.error("TODAY-004 : 투데이 게시글 생성 에러 - $e, $uid")
                throw CommonException("TODAY-004", HttpStatus.BAD_REQUEST)
            }

            // 보드 저장
            boardRepository.save(aBoard)

            for (file in files) {

                // Temp 파일 생성
                val tempFile = File.createTempFile(System.getProperty("java.io.tmpdir") + System.getProperty("file.separator") +  files[0].originalFilename, "");
                // 시스템 프로퍼티 경로로 파일 전송
                file.transferTo(tempFile)

                // MIME 타입 식별을 위해 FileTypeDetector 이용
                var bufferedStream = BufferedInputStream(tempFile.inputStream())
                var extension = FileTypeDetector.detectFileType(bufferedStream);

                // 이미지 확장자 검증
                if (ncpS3Uploader.checkImageExtension(extension) == null) {
                    log.error("S3-005 : 이미지 파일이 아닙니다")
                    throw CommonException("S3-005", HttpStatus.BAD_REQUEST)
                }

                // 이미지 랜덤 파일명, 메타데이터 생성
                var fileName : String = UUID.randomUUID().toString().plus(".").plus(extension.commonExtension)
                var albumFileLocation : String = ncpS3Uploader.fileLocation("album/${anAlbum.id}", fileName)
                var boardFileLocation : String = ncpS3Uploader.fileLocation("board/${aBoard.id}", fileName)

                var metaData : ObjectMetadata = ObjectMetadata()
                metaData.contentLength = tempFile.length()
                metaData.contentType = extension.mimeType

                // 앨범 이미지 파일 업로드 로직 (분리 필요)
                try {
                    ncp.assetNcpS3Client().putObject(
                            PutObjectRequest(
                                    bucketName,
                                    albumFileLocation,
                                    tempFile.inputStream(),
                                    metaData
                            )
                    )

                } catch (e : Exception) {
                    log.error("에러 발생")
                    throw CommonException("에러다!", HttpStatus.BAD_REQUEST)
                }

                // 보드 이미지 파일 업로드 로직 (분리 필요)
                try {
                    ncp.assetNcpS3Client().putObject(
                            PutObjectRequest(
                                    bucketName,
                                    boardFileLocation,
                                    tempFile.inputStream(),
                                    metaData
                            )
                    )
                } catch (e : Exception) {
                    log.error("에러 발생")
                    throw CommonException("에러다!", HttpStatus.BAD_REQUEST)
                }

                // 리스트에 추가
                albumImages.add("album/${anAlbum.id}/${fileName}")
                boardImages.add("board/${aBoard.id}/${fileName}")

            }

            // 앨범 이미지 업데이트
            try {
                anAlbum.changeImages(albumImages)
            } catch (e: CommonException) {
                log.error("ALBUM-IMG-002 : 앨범 이미지 수정 실패 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id}, $e ]")
                anAlbum.deleteAlbum()
                throw CommonException("ALBUM-IMG-002", HttpStatus.BAD_REQUEST)
            }

            // 보드 이미지 업데이트
            try {
                aBoard.updatePics(boardImages)
            } catch (e: CommonException) {
                log.error("ALBUM-IMG-002 : 앨범 이미지 수정 실패 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id}, $e ]")
                anAlbum.deleteAlbum()
                throw CommonException("ALBUM-IMG-002", HttpStatus.BAD_REQUEST)
            }

            log.info("투데이펫 / 앨범 동시 업로드 완료 album : ${anAlbum.id} / board : ${aBoard.id}")
        }

        else {
            val albumImages = try {
                ncpS3Uploader.uploadImages(files, "album/${anAlbum.id}");
            } catch (e: CommonException) {
                anAlbum.deleteAlbum()
                log.error("ALBUM-IMG-001 : 앨범 이미지 업로드 실패 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id}, $e ]")
                throw CommonException("ALBUM-IMG-001", HttpStatus.BAD_REQUEST)
            }

            try {
                anAlbum.changeImages(albumImages)
            } catch (e: CommonException) {
                log.error("ALBUM-IMG-002 : 앨범 이미지 수정 실패 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id}, $e ]")
                anAlbum.deleteAlbum()
                throw CommonException("ALBUM-IMG-002", HttpStatus.BAD_REQUEST)
            }

            log.info("앨범 생성 완료 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id} ]")
        }



        return AlbumDTO.ModelMapper.entityToDto(anAlbum, false)
    }
```

### NcpS3Uploader

```kotlin
    /**
     * @param files : 이미지 파일
     * @param boardId : 게시판 번호
     * @param albumId : 앨범 번호
     * @decc 앨범 & 보드 이미지 동시 업로드 함수
     */
    fun uploadAlbumWithBoard(files: List<MultipartFile>, boardId: String, albumId: String): MutableList<String> {

        val results = mutableListOf<String>()

        for (file in files) {

            val tempFile = File.createTempFile(
                System.getProperty("java.io.tmpdir") +
                        System.getProperty("file.separator") +
                        file.originalFilename, "")
            file.transferTo(tempFile)

            var bufferedStream = BufferedInputStream(tempFile.inputStream())
            val extension = detectFileType(bufferedStream)

            if (checkImageExtension(extension) == null) {
                log.error("S3-005 : 이미지 파일이 아닙니다")
                throw CommonException("S3-005", HttpStatus.BAD_REQUEST)
            }

            val locations = listOf(
                fileLocation("album/${boardId}", generateFileName(extension)),
                fileLocation("board/${albumId}", generateFileName(extension))
            )

            var metaData = ObjectMetadata()
            metaData.contentLength = tempFile.length()
            metaData.contentType = extension.mimeType

            for (location in locations) {
                try {
                    uploadToS3(bucketName!!, location, tempFile, metaData)
                } catch (e: Exception) {
                    if (results.isNotEmpty()) {
                        deleteFromS3(bucketName!!, location)
                    }
                    log.error("S3-004 : 이미지 업로드 실패, 버킷 이미지 삭제 - $e")
                    throw e
                }
            }

            results.addAll(locations.map { "$it/${tempFile.name}" })
        }

        log.info("NCP S3 앨범 & 보드 동시 이미지 업로드 성공 - ${results[0]}, ${results[1]}")

        return results
    }

    /** @desc --- uploadAlbumWithBoard를 위한 함수들 --- */
    fun generateFileName(extension: FileType): String {
        return UUID.randomUUID().toString().plus(".").plus(extension.commonExtension)
    }

    fun uploadToS3(bucketName: String, fileLocation: String, tempFile: File, metaData: ObjectMetadata) {
        try {
            ncp.assetNcpS3Client().putObject(
                PutObjectRequest(bucketName, fileLocation, tempFile.inputStream(), metaData)
            )
        } catch (e: Exception) {
            deleteFromS3(bucketName, fileLocation)
            log.error("S3-004 : S3 업로드 실패 - $e")
            throw CommonException("S3-004", HttpStatus.BAD_REQUEST)
        }
    }

    fun deleteFromS3(bucketName: String, fileLocation: String) {
        try {
            ncp.assetNcpS3Client().deleteObject(DeleteObjectRequest(bucketName, fileLocation))
        } catch (e: Exception) {
            log.error("S3-003 : S3 오브젝트 삭제 실패 - $e")
            throw e
        }
    }
```

```kotlin
@Transactional
    fun postAlbum (files : List<MultipartFile>, body : AlbumPostDTO, uid : String) : AlbumDTO {

        var anUser : User = userRecycleFn.checkActiveUser(uid)

        var anAlbum : Album = try {
            postAlbumEntity(body, uid)
        } catch (e : Exception) {
            log.error("ALBUM-003 : 앨범 생성 실패 - [ 유저명 : $uid, $e ]")
            throw CommonException("ALBUM-003", HttpStatus.BAD_REQUEST)
        }

        if (body.todayPet == true) {
            val aBoard: Board = try {
                Board(user = anUser).apply {
                    kind = "일상"
                    requestContent = body.content
                    createdAt = LocalDateTime.now()
                    pushUsers = mutableListOf(uid)
                }
            } catch (e: Exception) {
                log.error("TODAY-004 : 투데이 게시글 생성 에러 - $e, $uid")
                throw CommonException("TODAY-004", HttpStatus.BAD_REQUEST)
            }

            try {
                boardRepository.save(aBoard)
            } catch (e: Exception) {
                log.error("ALBUM-003 : 앨범 저장 실패 - [ 유저명 : $uid ] , $e")
                throw CommonException("ALBUM-003", HttpStatus.NOT_FOUND)
            }

            val albumImages= mutableListOf<String>()
            val boardImages = mutableListOf<String>()

            val lists = try {
                ncpS3Uploader.uploadAlbumWithBoard(files, aBoard.id.toString(), anAlbum.id.toString())
            } catch (e: Exception) {
                log.error("ALBUM-IMG-001 : 앨범 & 보드 이미지 동시 업로드 실패 - $e")
                throw CommonException("ALBUM-IMG-001", HttpStatus.BAD_REQUEST)
            }

            try {
                lists.filter { it.contains("album") }.map { albumImages.add(it) }
                lists.filter { it.contains("board") }.map { boardImages.add(it) }
            } catch (e: Exception) {
                log.error("ALBUM-003 : 업로드 이미지 리스트 변환 실패 - $e")
                throw CommonException("ALBUM-003", HttpStatus.NOT_FOUND)
            }

            try {
                anAlbum.changeImages(albumImages)
                aBoard.updatePics(boardImages)
            } catch (e: CommonException) {
                log.error("ALBUM-IMG-002 : 앨범 이미지 수정 실패 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id}, $e ]")
                anAlbum.deleteAlbum()
                aBoard.removePics()
                throw CommonException("ALBUM-IMG-002", HttpStatus.BAD_REQUEST)
            }

            log.info("투데이펫 / 앨범 동시 업로드 완료 album : ${anAlbum.id} / board : ${aBoard.id}")
        }


        else {
            val albumImages = try {
                ncpS3Uploader.uploadImages(files, "album/${anAlbum.id}");
            } catch (e: CommonException) {
                anAlbum.deleteAlbum()
                log.error("ALBUM-IMG-001 : 앨범 이미지 업로드 실패 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id}, $e ]")
                throw CommonException("ALBUM-IMG-001", HttpStatus.BAD_REQUEST)
            }

            try {
                anAlbum.changeImages(albumImages)
            } catch (e: CommonException) {
                log.error("ALBUM-IMG-002 : 앨범 이미지 수정 실패 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id}, $e ]")
                anAlbum.deleteAlbum()

            }

            log.info("앨범 생성 완료 - [ 유저명 : $uid, 앨범 번호 : ${anAlbum.id} ]")
        }

        return AlbumDTO.ModelMapper.entityToDto(anAlbum, false)
    }
```
