## 요구사항

- 유저들이 fcm 푸시 메시지를 받았을 경우 클릭한 횟수를 조회하는 MessageAnalyze 엔티티를 새로 생성했다.
- 이를 어드민 단계에서 이용하고 시각화 하기 위한 api가 필요하다.
- 다양한 어드민 페이지 혹은 데이터 시각화 툴 (엘라스틱 서치등)을 참고하여 어드민 페이지에서 사용할 데이터 dto 포맷을 결정한다

<br>

1. 전체 MessageAnalyze를 검색할 Get 검색 API
   - 페이징이 들어갈 것
   - 검색 기능이 들어갈 것 (다양한 포맷 / 날짜별 검색 / 타이틀 검색 등)
   - 어드민만 접근해야 할 것 
2. 상세 MessageAnalyze를 시각화 하기 위한 Get API
   - 어드민만 접근해야 할 것
3. MessageAnalyze의 ID는 UUID값으로 설정 (8자리)
   
<br>

**생각해보기**

1. 시작날짜만 들어온 경우 시작 날짜로부터 그 후까지의 데이터 반환

2. 종료날짜만 들어온 경우 종료 날짜로부터 그 후까지의 데이터 반환

3. 시작, 종료 날짜가 둘 다 들어왔을때는 시작과 종료 사이의 데이터 반환

## QueryDsl 동적 쿼리를 이용한 and & or 처리

- 동적 쿼리 작성 후, 현재 날짜 비교 로직 추가(Querydsl between)
- 날짜는 시작날짜, 종료날짜가 있으며, 시작날짜만 입력 시, 시작날짜부터 뒤로 리스트화, 종료도 마찬가지
- 시작과 종료날짜 둘 다 있을 경우, 사이의 메시지들을 리스트화 하여 반환
- BooleanExpression에 BooleanBuilder의 goe, loe 메서드 사용

**CustomMessageAnalyzeRepositoryImpl**

```kotlin
/**  
 * @author 신건우  
 * @desc Message Analyze 클릭 데이터 조건 검색 어드민 기능  
 */  
class CustomMessageAnalyzeRepositoryImpl(  
    @Autowired private val queryFactory: JPAQueryFactory  
) : QuerydslRepositorySupport(MessageAnalyze::class.java), CustomMessageAnalyzeRepository {  
      
    /**  
     * @author 신건우  
     * @desc Create Query Factory  
     */    
     override fun findByCondition(  
        data: MessageAnalyzeDTO.Post,  
    ): MutableList<MessageAnalyze>? {  
  
        val paging = PageRequest.of(0, 30)  
  
        val builder = data?.let { searchCondition(it) }  
  
        return queryFactory  
            .selectFrom(messageAnalyze)  
            .where(builder)  
            .orderBy(messageAnalyze.createdAt.desc())  
            .offset(paging.offset)  
            .limit(30.toLong())  
            .fetch()  
    }  
  
    /**  
     * @author 신건우  
     * @desc QueryDsl - 조건 검색 BooleanBuilder     */    
     fun searchCondition(  
        data: MessageAnalyzeDTO.Post,  
    ): BooleanBuilder? {  
  
        val builder = BooleanBuilder()  
        val startDate = data.start?.let { LocalDateTime.of(it.year, it.month, it.dayOfMonth, 0, 0 ,0) }  
        val endDate = data.end?.let { LocalDateTime.of(it.year, it.month, it.dayOfMonth, 23 ,59, 59)}  
  
        fun eqId(id: String): BooleanExpression? {  
            return if (id.isEmpty()) null else messageAnalyze.id.contains(id)  
        }  
  
        fun eqTitle(title: String): BooleanExpression? {  
            return if (title.isEmpty()) null else messageAnalyze.title.contains(title)  
        }  
  
        fun eqBody(body: String): BooleanExpression? {  
            return if (body.isEmpty()) null else messageAnalyze.body.contains(body)  
        }  
  
        fun eqType(type: String): BooleanExpression? {  
            return if (type.isEmpty()) null else messageAnalyze.type.contains(type)  
        }  
  
        fun eqCreatedAt(createdAt: LocalDateTime): BooleanExpression? {  
            return messageAnalyze.createdAt.eq(createdAt)  
        }  
  
        if (data.and!!) builder.and(data.id?.let { eqId(it) }) else builder.or(data.id?.let { eqId(it) })  
  
        if (data.and!!) builder.and(data.title?.let { eqTitle(it) }) else builder.or(data.title?.let { eqTitle(it) })  
  
        if (data.and!!) builder.and(data.body?.let { eqBody(it) }) else builder.or(data.body?.let { eqBody(it) })  
  
        if (data.and!!) builder.and(data.type?.let { eqType(it) }) else builder.or(data.type?.let { eqType(it) })  
  
        if (data.and!!) builder.and(data.createdAt?.let { eqCreatedAt(it) }) else builder.or(data.createdAt?.let { eqCreatedAt(it) })  
  
        if (data.start != null && data.end != null) {  
            builder.and(messageAnalyze.createdAt.between(startDate, endDate))  
        } else if (data.start != null) {  
            builder.and(messageAnalyze.createdAt.goe(startDate))  
        } else if (data.end != null) {  
            builder.and(messageAnalyze.createdAt.loe(endDate))  
        }  
  
        return builder  
    }  
}
```

## DTO & Entity

**MessageAnalyzeDTO**

```kotlin
/**  
 * @author 신건우  
 * @desc Message Analyze DTO  
 * @desc id 값은 UUID 값으로 받는다.  
 */class MessageAnalyzeDTO {  
  
    @JsonInclude(JsonInclude.Include.NON_NULL)  
    data class Post(  
        var id: String? = "",  
        var title: String? = "",  
        var body: String? = "",  
        var type: String? = "",  
        var data: String? = "",  
        var start: LocalDate? = null,  
        var end: LocalDate? = null,
        var createdAt: LocalDateTime? = null   
        var and: Boolean? = true,   
    ) {  
        object ModelMapper {  
  
            fun entityToDto(form: MessageAnalyze): Post {  
                return Post(  
                    id = form.id,  
                    title = form.title,  
                    body = form.body,  
                    type = form.type,  
                    data = form.data,  
                    createdAt = form.createdAt  
                )  
            }  
        }  
    }  
  
    @JsonInclude(JsonInclude.Include.NON_NULL)  
    data class Detail(  
        var id: String? = "",  
        var title: String? = "",  
        var body: String? = "",  
        var type: String? = "",  
        var data: String? = "",  
        var send: Int? = 0,  
        var reach: Int? = 0,  
        var click: Int? = 0,  
        var createdAt: LocalDateTime? = null,  
    ) {  
        companion object {  
            fun fromEntity(form: MessageAnalyze): Detail {  
                return Detail(  
                    id = form.id,  
                    title = form.title,  
                    body = form.body,  
                    type = form.type,  
                    data = form.data,  
                    send = form.send,  
                    reach = form.reach,  
                    click = form.click,  
                    createdAt = form.createdAt  
                )  
            }  
        }  
    }  
}
```

## Service

**MessageService**

```kotlin
/**  
 * @author 신건우  
 * @param mId : 메시지 ID * @desc Message Analyze 단건 상세조회 Admin 기능  
 */  
fun getMessageAnalyzeDetail(mId: String): MessageAnalyzeDTO.Detail? {  
  
    val analyze = try {  
        messageAnalyzeRepository.findById(mId).get()  
    } catch (e: Exception) {  
        log.error("Message Analyze 조회 실패 - 메시지 ID - $mId")  
        throw CommonException("Message Analyze 조회 실패", HttpStatus.NOT_FOUND)  
    }  
  
    val result = MessageAnalyzeDTO.Detail.fromEntity(analyze)  
    log.info("Message Analyze 조회 성공 - 메시지 ID : $mId")  
  
    return result  
}  
  
/**  
 * @author 신건우  
 * @param data : Message Analyze 데이터  
 * @desc Message Analyze 전체 조건 검색 Admin 기능 (페이지네이션)  
 */fun getAllMessageAnalyze(  
    data: MessageAnalyzeDTO.Post?  
): List<MessageAnalyzeDTO.Detail>?  {  
  
    val list = try {  
        data?.let { messageAnalyzeRepository.findByCondition(it) }  
    } catch (e: Exception) {  
        log.error("Message Analyze 전체 조회 실패 - $e")  
        throw CommonException("Message Analyze 전체 조회 실패", HttpStatus.BAD_REQUEST)  
    }  
  
    log.info("Message Analyze 전체 조건 검색 성공 - 메시지 ID : ${data?.id}")  
  
    return list?.stream()?.map { MessageAnalyzeDTO.Detail.fromEntity(it) }?.toList()  
}
```
