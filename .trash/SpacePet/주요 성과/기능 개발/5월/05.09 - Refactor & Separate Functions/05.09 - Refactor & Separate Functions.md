## 요구사항

1. 기능 별 함수 분리
2. 중복 메서드 병합 or Recycle 화
3. Injection된 Dependency끼리 최대한 같이 분리
4. 대상 클래스 : UserService, ProfileService
5. ProfileService의 모든 함수 → User로 이동
6. 누락된 에러 로그 작성 (log.info, log.error, log.warn)
7. 함수 성능 코드 개선
8. 변수명 재 정의 & 통일 (ex: userId → uid,  user → anUser,  유저uid → 유저 아이디(log))
9. 불필요하게 명시된 반환타입 삭제, 카멜 케이스 적용 안된것들 수정

    suspend fun getChatRoomCountData(uid: String): Boolean {  
  
        var ref = FirebaseDatabase.getInstance()  
        val childRef: DatabaseReference = ref.getReference("users/${uid}/")  
  
        val data = suspendCoroutine<Any?> {  
            childRef.addListenerForSingleValueEvent(object : ValueEventListener {  
                override fun onDataChange(dataSnapshot: DataSnapshot) {  
                    it.resume(dataSnapshot.value)  
                }  
  
                override fun onCancelled(databaseError: DatabaseError) {}  
            })  
        }  
  
        var json = JsonParser.parseString(Gson().toJson(data)).asJsonObject  
        var chatRoomSize = json.keySet().size  
  
        if (chatRoomSize >= 2) {  
            return true  
        }  
        return false  
    }

    fun scheduledSaveICR() {  
  
        var now = LocalDateTime.now()  
  
        var day01Date = now.minusDays(2)  
        var day02Date = now.minusDays(3)  
        var day03Date = now.minusDays(4)  
  
        var ICRD1 = getICR(day01Date)  
        var ICRD2 = getICR(day02Date)  
        var ICRD3 = getICR(day03Date)  
  
        var dateFormat = DateTimeFormatter.ofPattern("YYYY-MM-dd")

        var day02String = day02Date.format(dateFormat)  
        var day03String = day03Date.format(dateFormat)  
  
        var day01 = icrRepository.findByDate(day01String)  
        var day02 = icrRepository.findByDate(day02String)  
        var day03 = icrRepository.findByDate(day03String)  
  
        if (day01.isEmpty()) {  
            icrRepository.save(  
                Icr(  
                    id = UUID.randomUUID().toString(),  
                    day = day01String,  
                    d1 = ICRD1.toString(),  
                    createdAt = LocalDateTime.now()  
                )  
            )  
        } else {  
            day01[0].updateD1(ICRD1.toString())  
        }  
  
        if (day02.isEmpty()) {  
            icrRepository.save(  
                Icr(  
                    id = UUID.randomUUID().toString(),  
                    day = day02String,  
                    d2 = ICRD2.toString(),  
                    createdAt = LocalDateTime.now()  
                )  
            )  
        } else {  
            day02[0].updateD2(ICRD2.toString())  
        }  
  
        if (day03.isEmpty()) {  
            icrRepository.save(  
                Icr(  
                    id = UUID.randomUUID().toString(),  
                    day = day03String,  
                    d3 = ICRD3.toString(),  
                    createdAt = LocalDateTime.now()

            )

    }  
  
    /**  
     * @author 고범석  
     * @param data : 가입 날짜  
     * @desc get icr  
     */    
     fun getICR(date: LocalDateTime): Int {  
  
        // ICR Day1  
        // 1. 24시간 기준 가입 회원 (정회원)  
        var dayStart = LocalDateTime.of(LocalDate.of(date.year, date.monthValue, date.dayOfMonth), LocalTime.of(0, 0))  
        var dayEnd = LocalDateTime.of(LocalDate.of(date.year, date.monthValue, date.dayOfMonth), LocalTime.of(23, 59))  
  
        var users: List<User> = try {  
            userRepository.findByNewUser(dayStart, dayEnd)  
        } catch (e: Exception) {  
            throw CommonException("유저 검색 실패", HttpStatus.BAD_REQUEST)  
        }  
  
        // 2-1. 좋아요를 누른 사람  
        // 2-2. 채팅방이 어드민 채팅방 이외에 있는 사람  
        // 2-3. 이웃의 한마디를 남긴 사람  
        // 2-4. 게시글 댓글을 남긴 사람  
        var result = 0  
  
        for (user in users) {  
            var likes = userLikeService.getICRLike(user.id)  
            var chatRoom: Boolean = false  
            runBlocking {  
                chatRoom = getChatRoomCountData(user.id)  
            }  
            var neighbors = commentService.getPostNeighborCommentCount(user.id) >= 1  
            var boards = commentService.getPostBoardCommentCount(user.id) >= 1  
  
            if (likes || chatRoom || neighbors || boards) {  
                result += 1  
            }  
        }  
        return result  
    }

                    log.info("1 DAY Retention 측정 - $uid")  
                }  
            }  
        } catch (e: Exception) {  
            log.info("1 DAY Retention 측정 실패 - $uid")  
        }  
  
        // 3 Day Retention 측정  
        try {  
            if (anUser.userDate?.threeDayRetention == null) {  
  
                var threeDayDuration = createdAt?.plusDays(3)  
                var date1 = threeDayDuration?.truncatedTo(ChronoUnit.DAYS)  
                var date2 = now.truncatedTo(ChronoUnit.DAYS)  
                var check = date2.compareTo(date1)  
  
                if (check in -2..1) {  
                    anUser.userDate?.updateThreeDayRetention(nowDate)  
                    log.info("3 DAY Retention 측정 - $uid")  
                }  
            }  
        } catch (e: Exception) {  
            log.info("3 DAY Retention 측정 실패 - $uid")
        }  
  
        // 7 Day Retention  
        try {  
            if (anUser.userDate?.sevenDayRetention == null) {  
  
                var sevenDayDuration = createdAt?.plusDays(7)  
                var date1 = sevenDayDuration?.truncatedTo(ChronoUnit.DAYS)  
                var date2 = now.truncatedTo(ChronoUnit.DAYS)  
                var check = date2.compareTo(date1)  
  
                if (check in -6..1) {  
                    anUser.userDate?.updateSevenDayRetention(nowDate)  
                    log.info("7 DAY Retention 측정 - $uid")  
                }  
            }  
        } catch (e: Exception) {  
            log.info("7 DAY Retention 측정 실패 - $uid")  
        }  
    }  
}

 * @desc 유저 인증 정보 생성  
 */  
@Transactional  
fun getUserCert(uid : String) : UserCert {  
  
    val anUser : User = checkActiveUser(uid)  
    val anUserCert : UserCert? = userCertRepository.findByUserId(uid)  
  
    if (anUserCert == null) {  
  
        var userCert = UserCert().apply {  
            imageCert = anUser.gradeAuth?.authentication != "0"  
            imageCertAt = if (anUser.gradeAuth?.authentication == "1") anUser.userDate?.gradeAt else null  
        }  
  
        try {  
            anUser.addUserCert(userCert)  
            userCertRepository.save(userCert)  
        } catch (e: Exception) {  
            log.error("USER-100 : UserCert 업데이트 실패 [유저 아이디 : $uid] : $e")  
            throw CommonException("USER-002", HttpStatus.EXPECTATION_FAILED)  
        }  
  
        log.info("유저 인증 생성 후 제공 완료 [ 유저 아이디 : $uid ]")  
  
        return userCert  
    }  
    else {  
        if (!anUserCert.imageCert && anUser.gradeAuth?.authentication != "0") {  
            anUserCert.adminPatchUserImage(true)  
        }  
  
        log.info("유저 인증 제공 완료 [ 유저 아이디 : $uid ]")  
  
        return anUserCert  
    }  
}
