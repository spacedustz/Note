## 요구사항

대상 서비스 컨트롤러 GET `/neighbor/new`

기존 DTO

`NeighborListDTO` 에 조회한 유저 기준으로 Boolean? myTown ,Boolean? myNear 를 추가하고자 한다.

<br>

이 때 조건은 다음과 같다

-   myNear (내 주변 시군구에 포함될 경우 true)
    
-   myTown (나와 시군구까지 같을 경우 true 나머지는 null)
    
-   전체 컨트롤러의 유저 수는 50명으로 변경 (페이징) (단 첫페이지만 반환하면 되므로 페이징 대신 limit가 어울릴듯)
    
-   기존 시군구 기반 검색은 무시 (전체 지역으로 검색)
    
-   정회원, 유저 가입순, 50 limit, No Pagination
    
-   Geometry Service의 함수 참조

<br>

### 생각해보기

- 서비스에서 조회유저와 유저 리스트의 Geometry 비교

```kotlin
/**  
 * @author 신건우  
 * @param notInUsers : 제외 유저  
 * @desc 유저 50명 검색  
 */  
override fun findFiftyUser(notInUsers: MutableList<String>): List<User> {  
    return from(user)  
        .where(  
            user.grade.eq("정회원")  
                .and(user.deleted.eq(false))  
                .and(user.id.notIn(notInUsers))  
        )  
        .limit(50)  
        .orderBy(user.userDate.createdAt.desc())  
        .fetch()  
}
```

```kotlin
fun entityToRecommendGeometryDTO(form: User, userLikes : List<UserLike>?, geometry: List<GeometryDTO.SQLGeometry>, userSigungu: String) : NeighborListDTO {  
  
    var neighborKeyword = form.userDetail?.neighborKeyword ?: listOf()  
    var userLiked = false  
  
    if (userLikes != null && userLikes.firstOrNull{it -> it.taker?.id == form.id} != null) {  
        userLiked = true  
    }
  
    var near: Boolean? = false  
    var town: Boolean? = false  
  
    val sigungu = form.geometry?.sigungu  
    val containsSigungu: Boolean = geometry.any { it.sigungu == sigungu }  
    val matchSigungu: Boolean = geometry.any{ it.sigungu == userSigungu && sigungu == userSigungu }  
  
    if (containsSigungu) near = true  
    if (matchSigungu) town = true  
  
        return NeighborListDTO(  
            uid = form.id,  
            userName = form.name,  
            userGender = form.gender,  
            userAgeRange = form.ageRange,  
            userGrade = form.grade,  
            animal = form.animals[0].animal,  
            animalName = form.animals[0].name,  
            animalGender = form.animals[0].gender,  
            animalBirthDay = form.animals[0].birthday,  
            animalProfile = form.animals[0].profile,  
            neighborKeyword = if (neighborKeyword.isNotEmpty()) listOf(neighborKeyword[0]) else listOf(),  
            geometry = form.geometry?.let { LocationResDTO.ModelMapper.entityToDto(it) },  
            userGive = form.give,  
            userTake = form.take,  
            like = form.likeCount ?: 0,  
            mateKeyword = form.userDetail?.mateKeyword ?: listOf(),  
            wishToConnect = false,  
            wishType = "",  
            createdAt = form.userDate?.createdAt,  
            authentication = form.gradeAuth!!.authentication ?: "2",  
            wishNeighbor = listOf(),  
            trustPoint = form.trustPoint,  
            breed = form.animals[0].breed,  
            mateDetail = form.userDetail?.mateDetail,  
            userCert = UserCertDTO.ModelMapper.entityToDTOorNull(form.userCert),  
            userLiked = userLiked,  
            myNear = near,  
            myTown = town  
        )  
}
```

```kotlin
fun getNeighborNew(userId: String): List<NeighborListDTO> {  
  
    val anUser = userRecycleFn.checkActiveUser(userId)  
    val notInUsers = userRecycleFn.excludeUsers(mutableListOf(anUser.id))  
  
    if (anUser.blockedUsers != null) {  
        notInUsers.addAll(anUser.blockedUsers!!)  
    }  
  
    var geometrys = geometryService.getNearSigungu(anUser.geometry?.sido, anUser.geometry?.sigungu)  
  
    val users : List<User> = try {  
        userRepository.findFiftyUser(notInUsers)  
    } catch (e: Exception) {  
        log.error("BOARD-003 : SQL 유저 주변 이웃 조회 실패 [userId : ${anUser.id}] : $e")  
        throw CommonException("BOARD-003", HttpStatus.NOT_FOUND)  
    }  
  
    var takers : List<String> = users.stream().map{ it.id }.toList()  
  
    val userLikes : List<UserLike> = userLikeRecycleFn.findUserAlreadyLikes(userId, takers)  
  
    log.info("[ 이웃 조회 완료 ]  조회한 유저: ${anUser.id}, 조회한 유저의 시군구:  ${anUser.geometry?.sido} ${anUser.geometry?.sigungu} ${anUser.geometry?.dong}")  
    log.info("조회한 유저의 인근 시군구 : ${geometrys.map { it.sigungu }.toList()}")  
      
    return users.stream().map { NeighborListDTO.ModelMapper.entityToRecommendGeometryDTO(it, userLikes, geometrys, anUser.geometry?.sigungu ?: "") }.toList()  
}
```
