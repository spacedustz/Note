## 💡 요구사항
<br>
```markdown
- Community Board Service에서 진행

- 현재 API에서는 사진 업로드 따로 텍스트 업로드 따로로 이루어져 있음

- 두가지 API를 하나로 합칠것

- Post 메소드와 User만 접근 가능해야 함

- s3 메소드와 함께해요 업로드 메소드를 합쳐 새로운 api 생성

- 반환 값은 Board Detail 메소드와 동일해야 함 (상세 설명 할 예정)

- 이미지 업로드를 하는 경우도 / 안 하는 경우도 있음

- 중간에 이미지 업로드를 실패할 경우에도 게시글 자체는 업로드를 성공해야 할것

- 최소한의 추가 코드로 짜 볼 것 (이미 서로 다른 두 기능이 있음 복붙으로도 완성 가능)

- 모든 게시글 유형에 있어서 반영할 수 있도록 범용성있게 설계해 볼 것 (단 텍스트 포맷은 전부 다름)
```

### Controller

```kotlin
/**  
 * @author 신건우  
 * @param multipartFile : 이미지 파일  
 * @param data : 게시글 정보  
 * @param uid : 유저 아이디  
 * @desc 게시글 등록  
 */  
@Auth(role = USER)  
@PostMapping(consumes = [MediaType.APPLICATION_JSON_VALUE, MediaType.MULTIPART_FORM_DATA_VALUE])  
fun postBoard(  
        @RequestPart multipartFile: List<MultipartFile>,  
        @RequestPart data : BoardSaveReqDTO,  
        @RequestAttribute("uid") uid : String  
) : ResponseEntity<BoardResponseDTO> {  
    return ResponseEntity(communityService.postBoard(multipartFile, data, uid), HttpStatus.CREATED)  
}
```

### Service

```kotlin
/**  
 * @author 신건우  
 * @param multipartFile : 이미지 파일  
 * @data : 게시글 정보  
 * @param uid : 유저 아이디  
 * @desc 게시글 등록  
 */  
@Transactional  
fun postBoard(multipartFile: List<MultipartFile>, data: BoardSaveReqDTO, uid: String): BoardResponseDTO {  
  
    var anUser = userRecycleFn.checkActiveUser(uid)  
  
    if (anUser.grade != "정회원" && data.kind != "일상") {  
        log.error("앨범 업로드 에러 : 부회원 시도 $uid")  
        throw CommonException("BOARD-001 - 회원 등급 오류", HttpStatus.BAD_REQUEST)  
    }  
  
    val aBoard: Board = try {  
        Board(user = anUser).apply {  
            requestContent = data.requestContent  
            requestSecondContent = data.requestSecondContent  
            preMeeting = data.preMeeting  
            requestGender = data.requestGender  
            requestAnimalUser = data.requestAnimalUser  
            requestedAt = data.requestedDate  
            kind = data.kind  
            pushUsers = mutableListOf(uid)  
            timeHelp = data.timeHelp.toMutableList()  
            animalNetralization = null  
            animalSize = null  
            requestedAt02 = null  
            reword = null  
            timeReword = data.timeReword  
        }  
    } catch (e: Exception) {  
        throw CommonException("BOARD-007", HttpStatus.EXPECTATION_FAILED)  
    }  
  
    try {  
        boardRepository.save(aBoard)  
    } catch (e: Exception) {  
        log.error("BOARD-007 : 게시글 생성 실패 $e")  
        throw CommonException("BOARD-007", HttpStatus.EXPECTATION_FAILED)  
    }  
  
    if (multipartFile.isNotEmpty()) {  
        val aBoardImages = try {  
            ncpS3Uploader.uploadImages(multipartFile, "board/${aBoard.id}");  
        } catch (e: CommonException) {  
            aBoard.removePics()  
            log.error("")  
            throw CommonException("", HttpStatus.EXPECTATION_FAILED)  
        }  
        aBoard.updatePics(aBoardImages)  
    }  
  
    log.info("게시글 작성 완료  : [ 글 번호 : ${aBoard.id}, 글 작성자 : $uid ]")  
  
    return BoardResponseDTO.ModelMapper.entityToDto(aBoard)  
}
```
