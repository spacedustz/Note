## **💡** 요구사항
<br>
### **Comment Post API**

-   입력 Body
-   (모든 보드 공통 - 돌봄 게시글 포함)

```kotlin
var body = jsonEncode({
  "content" : content // content는 스트링
})
```

-   반환값으로 List Comment를 컴패니언화

```kotlin
class DTO (
  var comments : List<Comment>?
)
```

-   유저가 정회원인지 체크 / 50자 제한 체크 (제한 글자수 공유 예정)
-   댓글 알림 List String 에 댓글을 작성한 유저 uid가 없을 경우 추가 (connectedUsers 변수)

---

### **Comment Patch API**

-   입력 body 없음
-   파라미터로 보드 아이디 전달 및 status(boolean) 전달
-   Board Entity에 connectedUsers 유저 변수 사용 true status 일 경우 리스트에 uid 추가 false 일 경우 리스트에서 제거
-   성공일 경우 status를 그대로 반환

```kotlin
class DTO (
  var status : Boolean
)
```

---

### **Comment Delete API**

-   parameter Comment ID
-   댓글을 쓴 사람이 현재 삭제를 요하는 사람인지 확인 후 삭제처리 (delete만 1로 바꾸는 것)
-   반환값으로 List Comment를 컴패니언화

```kotlin
class DTO (
  var comments : List<Comment>?
)
```

### **Controller**

```kotlin
/**
     * @author 신건우
     * @param data : 댓글 정보
     * @param boardId : 게시글 번호
     * @param uid : 유저 아이디
     * @desc 댓글 작성 API
     */
    @Auth(role = USER)
    @PostMapping("/comments")
    fun postComment(

        @RequestParam("boardId", required = true) boardId: String,
        @RequestAttribute("uid") uid: String
    ): ResponseEntity<CommentListDTO> {
        return ResponseEntity(commentService.postComment(data, boardId, uid), HttpStatus.CREATED)
    }

    /**
     * @author 신건우
     * @param boardId : 게시글 번호
     * @param status : 댓글 알림 상태
     * @param uid : 유저 아이디
     * @desc 댓글 알림 수정 API
     */
    @Auth(role = USER)
    @PatchMapping("/comments/notice/fix")
    fun patchComment(
        @RequestParam("boardId", required = true) boardId: String,
        @RequestParam("status", required = true) status: Boolean,
        @RequestAttribute("uid") uid: String
    ): ResponseEntity<Boolean> {
        return ResponseEntity(commentService.patchComment(boardId, status, uid), HttpStatus.OK)
    }

    /**
     * @author 신건우
     * @param boardId : 게시글 번호
     * @param commentId : 댓글 번호
     * @param uid : 유저 아이디
     * @desc 댓글 삭제 API
     */
    @Auth(role = USER)
    @DeleteMapping("/comments/delete")
    fun deleteComment(
        @RequestParam("boardId", required = true) boardId: String,
        @RequestParam("commentId", required = true) commentId: Long,
        @RequestAttribute("uid") uid: String
    ): ResponseEntity<CommentListDTO> {
        return ResponseEntity(commentService.deleteComment(boardId, commentId, uid), HttpStatus.OK)
    }
```

### **Service**

```kotlin
/**
     * @author 신건우
     * @param data : 댓글 정보
     * @param boardId : 게시글 번호
     * @param uid : 유저 아이디
     * @desc 댓글 작성
     */
    @Transactional
    fun postComment(data: CommentReqDTO, boardId: String, uid: String): CommentListDTO {

        val anUser = userRecycleFn.checkActiveUser(uid)
        val aBoard = boardRecycleFn.checkBoardStatus(boardId.toLong(), uid, false)

        userRecycleFn.validateUser(anUser, "comment")

        try {
            if (!aBoard.connectedUsers.contains(uid)) {
                aBoard.connectedUsers.add(uid)
            }
        } catch (e: Exception) {
            log.error("COMMENT-004 - 댓글 알림 리스트에 유저 추가 실패 - [ 유저명 : $uid, 글번호: $boardId ]")
            throw CommonException("COMMENT-004", HttpStatus.BAD_REQUEST)
        }

        if (data.content.length > 50) {
            throw CommonException("댓글은 50글자만 작성할 수 있어요 - [ 글자 수 : ${data.content.length} ]", HttpStatus.BAD_REQUEST)
        }

        try {
            commentRepository.save(commentRecycleFn.createBoardComment(data, anUser, aBoard, 0))
        } catch (e: Exception) {
            log.error("COMMENT-003 : 댓글 생성 실패 - [ 유저명 : $uid, 글번호 : $boardId ]")
            throw CommonException("COMMENT-003", HttpStatus.BAD_REQUEST)
        }

        val list = try {
            CommentListDTO.ModelMapper.entityToDto(aBoard)
        } catch (e: Exception) {
            log.error("COMMENT-002 - 댓글 리스트 변환 실패 [ 유저명 : $uid, 글번호 : $boardId ]")
            throw CommonException("COMMENT-002", HttpStatus.NOT_FOUND)
        }

        log.info("댓글 작성 완료 - [ 유저명 : $uid, 글번호: $boardId ] ")

        return list
    }

    /**
     * @author 신건우
     * @param boardId : 게시글 번호
     * @param status : 댓글 알림 상태
     * @param uid : 유저 아이디
     * @desc 댓글 알림 리스트 수정
     */
    @Transactional
    fun patchComment(boardId: String, status: Boolean, uid: String): Boolean {

        val anUser = userRecycleFn.checkActiveUser(uid)
        val aBoard = boardRecycleFn.checkBoardStatus(boardId.toLong(), uid, false)

        userRecycleFn.validateUser(anUser, "comment")

        val result = try {
            if (status) {
                aBoard.connectedUsers.add(uid)
            } else {
                aBoard.connectedUsers.remove(uid)
            }
        } catch (e: Exception) {
            log.error("COMMENT-004 - 댓글 알림 리스트에 유저 추가 실패 - [ 유저명 : $uid, 글번호: $boardId ]")
            throw CommonException("COMMENT-004", HttpStatus.BAD_REQUEST)
        }

        log.info("댓글 알림 수정 완료 - [ 유저명 : $uid, 글번호: $boardId ] ")

        return result
    }

    /**
     * @author 신건우
     * @param boardId : 게시글 번호
     * @param commentId : 댓글 번호
     * @param uid : 유저 아이디
     * @desc 댓글 삭제
     */
    @Transactional
    fun deleteComment(boardId: String, commentId: Long, uid: String): CommentListDTO {
        val aBoard = boardRecycleFn.checkBoardStatus(boardId.toLong(), uid, false)
        val aComment = commentRecycleFn.checkActiveComment(commentId, uid)

        if (aComment.user!!.id == uid) aComment.deleted = true
        else throw CommonException("COMMENT-005 - 유저 정보가 맞지 않습니다. [ 유저명 : $uid ]", HttpStatus.FORBIDDEN)

        val list = try {
            CommentListDTO.ModelMapper.entityToDto(aBoard)
        } catch (e: Exception) {
            log.error("COMMENT-002 - 댓글 리스트 변환 실패 [ 유저명 : $uid, 글번호 : $boardId ]")
            throw CommonException("COMMENT-002", HttpStatus.NOT_FOUND)
        }

        log.info("댓글 삭제 완료 - [ 유저명 : $uid, 글번호: $boardId ] ")

        return list
    }
```

### **DTO**

```kotlin
/** 
 * @author 신건우
 * @desc 댓글 리스트 Companion화 
 */
data class CommentListDTO(
    var comments: List<CommentDTO>?
) {
    object ModelMapper {
        fun entityToDto(form: Board): CommentListDTO {
            return CommentListDTO(
                comments = form.comments.stream().filter { it.activateComment() }.map{CommentDTO.ModelMapper.entityToDto(it)}.toList()
            )
        }
    }
}
```
