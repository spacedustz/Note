## 🧑🏻‍💻 Version Check Function
---

## 💡 요구사항
<br>
### 입력값

- "ios" or "android"
<br>
### 최신 버전

- id : 215
- status : false
- version : 1.7.08
- Os : android
<br>
### 새로운 규칙

- Android - 홀수
- ios - 짝수
<br>
### 제한사항
- ios / android 두 개의 String 값만 들어오며 나머지는 Throw Exception
- 기존 DB에 없는 버전이 들어올 경우 db에 새로운 엔티티를 넣는 함수 개발
- 이 때, status는 항상 true로 저장해야 한다.
- 1.7.09 보다 1.7.10이 먼저 들어올 경우 ios의 id는 216이 아닌 218이 되어야 한다.

## 💡 구현
### HealthCheckService
- 버전 조회 기능 + 버전 변경 기능 개발
<br>
```kotlin
/**  
 * @author 신건우
 * @param v : 버전 정보  
 * @param o : OS 정보  
 * @desc 서버 버전 변경 기능  
 */  
fun changeVersion(  
    v : String,  
    o : String  
) : Version {  
  
    var newOs = when (o) {  
        "ios" -> o  
        "android" -> o  
        else -> throw CommonException("SERVER-004 : 플랫폼 에러, Android 나 iOS가 아닙니다. [ 입력된 값 : $o ]", HttpStatus.BAD_REQUEST)  
    }  
  
    // 1 7 10  
    var versionString = v.split(".")  
  
    if (versionString.size != 3) {  
        throw CommonException("SERVER-004 : 유효한 버전이 아닙니다. [ 입력된 값 : $versionString ]", HttpStatus.BAD_REQUEST)  
    }  
  
    // 타입 체크  
    for (i in versionString) {  
        if (i.toIntOrNull() == null) {  
            throw CommonException("SERVER-002 : 버전은 숫자만 가능합니다.", HttpStatus.NOT_FOUND)  
        }  
    }  
  
    // 1, 7, 08 - 1708  
    var example = 1708  
  
    // 1.7.09 보다 1.7.10이 먼저 들어올 경우 ios의 id는 216이 아닌 218이 되어야 한다  
    var newVersion =  
        versionString[0].toInt() * 1000 +  
                versionString[1].toInt() * 100 +  
                versionString[2].toInt()  
  
    var diff = newVersion - example  
  
    // 4 + 214 = 218 (ios 기본값)  
    var idString = (diff * 2) + 214  
  
    if (newOs == "android") {  
        idString += 1  
    }  
  
    // new Version  
    val newVer = Version().apply {  
        id = idString.toLong()  
        version = v  
        os = newOs  
        status = true  
    }  
  
    log.info("서버 버전 변경 완료 - [ 버전 번호 : ${newVer.id}, 플랫폼 : ${newVer.os}, 버전 상태 : ${newVer.status} ]")  
  
    return newVer  
}


/**  
 * @author 신건우
 * @param v : 버전 정보  
 * @param o : OS 정보  
 * @desc 서버 버전 조회 기능  
 */  
fun getVersionAndOs(version: String, os: String): Boolean {
        val versionInfo = try {
            versionRepository.search(version, os)
        } catch (e: Exception) {
            log.error("SERVER-004 : SQL 서버 버전 정보 조회 실패  : $e")
            throw CommonException("SERVER-004", HttpStatus.EXPECTATION_FAILED)
        }

        if (versionInfo == null) {
            val newVersion = try {
                versionRepository.save(changeVersion(version, os))
            } catch (e : Exception) {
                log.error("SERVER-005 : 존재하지 않는 버전입니다.")
                throw CommonException("SERVER-005", HttpStatus.NOT_FOUND)
            }

            return newVersion.status
        }

        log.info("서버 버전 정보 [ ${versionInfo.toString()} ]")

        return versionInfo.status
    }
```

### HealthCheckController
```kotlin
/**   
* @author 신건우  
 * @param version : 버전 정보  
 * @param os : os 정보  
 * @desc 유저 앱 버전 및 os 확인   
*/  
@GetMapping("/version")  
fun getVersionAndOs(  
    @RequestParam(name = "v", required = true) version: String,  
    @RequestParam(name = "os", required = true) os: String,  
): ResponseEntity<Boolean> {  
    return ResponseEntity(healthCheckService.getVersionAndOs(version, os), HttpStatus.OK)  
}
```

### Version Entity
```kotlin
@Entity  
@DynamicUpdate  
@DynamicInsert  
class Version(  
    @Id  
    var id: Long? = null,  
    var version: String = "",  
    var os: String = "",  
    var status: Boolean = true  
) {  
    override fun toString(): String {  
        return "Version(version='$version', os=$os, status=$status)"  
    }  
  
    fun modifyingVersionStatus(data : Boolean) {  
        this.status = data  
    }  
}
```
