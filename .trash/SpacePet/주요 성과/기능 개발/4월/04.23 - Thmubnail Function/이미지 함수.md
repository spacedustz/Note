fun getFileExtension(fileName: String): String {
    try {
        return fileName.substring(fileName.lastIndexOf(".") + 1)
    } catch (e: StringIndexOutofBoundsException) {
        log.error("")
        throw CommonException("", HttpStatus.BAD_REQUEST)
    }
}

fun convertMultipartFileToInputStream(file: MultipartFile): InputStream {
    try {
        return BufferedInputStream(file.inputStream)
    } catch (e: IOException) {
        log.error("")
        throw Exception("", HttpStatus.EXPECTATION_FAILED)

### 이미지 MetaData

```kotlin
fun imageMetadata(file: MultipartFile): ObjectMetadata {
    val metadata = ObjectMetadata()
    metadata.contentLength = file.size
    metadata.contentType = "image/${getFileExtension(file.originalFilename.toString())}"
    
    return metadata
}

/** @desc 이미지 업로드 */
    fun uploadImages(files: List<MultipartFile>, folderName: String = ""): MutableList<String> {

        if (files.isEmpty()) {
            log.error("S3-001 : 업로드하려는 파일이 비었습니다.")
            throw CommonException("S3-001", HttpStatus.BAD_REQUEST)
        }

        val result: MutableList<String> = mutableListOf()

        for (file in files) {
            if (file.originalFilename == null) {
                log.error("S3-002 : 이미지 이름이 없어요.")
                throw CommonException("S3-002", HttpStatus.BAD_REQUEST)
            }

            val fileObj: InputStream = convertMultiPartFileToInputStream(file)
            val fileName: String = createFileName(file.originalFilename.toString())

            try {
                ncp.assetNcpS3Client().putObject(
                    PutObjectRequest(
                        bucketName,
                        fileLocation(folderName, fileName),
                        fileObj,
                        imageMetadata(file)
                    )
                )
                result.add(fileLocation(folderName, fileName))
            } catch (e: Exception) {
                log.error("S3-004 : S3 업로드 실패 : $e")
                throw CommonException("S3-004", HttpStatus.BAD_REQUEST)
            }
        }

        log.info("이미지 업로드 성공 [ s3 디렉토리 이름 : $folderName ]")

        return result
```

 /** @desc 반려동물 사진 업데이트 */
    @Transactional
    fun updateAnimalImages(multipartFile: MultipartFile, userId: String): String {

        // 폴더 이름 반환
        fun makeFolderName(): String {
            return "user/".plus(userId).plus("/animalImages")
        }

        val anUser = userRecycleFn.checkActiveUser(userId)
        val animalImages = anUser.userDetail?.animalImages ?: listOf()

        if (anUser.userDetail == null || animalImages.size >= 12 ) {
            log.error("PROFILE-001 : User Detail 없는 유저 이미지 업로드 시도 || 12장 이상 업로드 시도")
            throw CommonException("PROFILE-001", HttpStatus.NOT_FOUND)
        }

        val imageURL = try {
            uploadImages(listOf(multipartFile), makeFolderName())
        } catch (e: Exception) {
            log.error("ANI-0003 :  이미지 업로드 실패 : $e")
            throw CommonException("ANI-0003", HttpStatus.BAD_REQUEST)
        }

        var updateAnimalImages = try {
            if (anUser.userDetail?.animalImages == null) {
                mutableListOf<String>()
            } else {
                anUser.userDetail?.animalImages!!.toMutableList()
            }
        } catch (e : Exception) {
            log.error("PROFILE-001 : 동물 이미지 배열 리스트 화 실패")
            throw CommonException("PROFILE-001", HttpStatus.NOT_FOUND)
        }

        try {
            updateAnimalImages.add(imageURL[0])
        } catch (e : Exception) {
            log.error("PROFILE-001 : 동물 이미지 배열 리스트 화 실패")
            throw CommonException("PROFILE-001", HttpStatus.NOT_FOUND)
        }

        try {
            anUser.userDetail?.updateAnimalImages(updateAnimalImages)
        } catch (e: Exception) {
            log.error("ANI-0004 : SQL 반려동물 프로필 이미지 업데이트 실패 : $e")
            throw CommonException("ANI-0004", HttpStatus.EXPECTATION_FAILED)
        }

        log.info("반려동물 프로필 사진 경로 :  ${imageURL[0]}")

        return imageURL[0]
    }
