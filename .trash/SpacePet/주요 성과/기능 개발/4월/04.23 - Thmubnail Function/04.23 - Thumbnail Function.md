## 요구사항

- 현재 보드 기능에 썸네일 기능을 추가해 원활한 서버 기능을 구현한다

- 멀티파트 파일을 Compress 하여 S3 (Object Storage)에 전송
  500KB 미만 파일은 압축하지 않는다
  500 KB ~ 1MB 파일은 500KB 미만으로 압축한다
  최대 500KB가 되도록 이미지 파일 크기 조정

  한장당 2MB 이상의 파일은 거부한다 (이상의 파일일 경우 App 에서 압축할 예정)
  이미지 포맷 고정 (PNG) - APP 단에서 변경 혹은 거부할 것

- 보드 이미지 원본과 썸네일 (압축된 파일)로 나눠 저장한다
  Board Entity images 속성에 첫번째 요소는 썸네일, 2번째 요소는 사진으로 표기
  (향후 대표 이미지 기능으로 활용할 수 있도록 정의한다)

- 파일 압축 및 S3 전송 후 서버에는 남는 파일이 없도록 해야한다
  파일 이름 규칙은 이미 정해진 것을 따른다

- API가 아닌 Function 구성으로 작성할 것 (공통 함수로 사용 예정)

  - ncpS3Uploader에 함수로 작성

- 여러 용량의 값이 들어옴 (300KB ~ 10MB 등)

  - 이때 결과값이 항상 300KB(대략)이 되기 위한 압축률을 Float 형태로 반환

### NcpS3Uploader

- AffineTransform() - 2D좌표로의 선형 매핑을 수행하는 2D 아핀 변환 클래스
  - 아핀 변환은 변환, 크기조정, 뒤집기, 회전 시퀀스를 사용하여 구성할 수 있음
  - scale()
  - translate()
  - AffineTransformOp
  
- ImageIO - 이미지 데이터 I/O를 할 수 있는 Python 라이브러리 (크로스 플랫폼)
  - read()
- ImageWriteParam - 스트림 인코딩 관련 클래스, 특정 이미지 형식을 위한 플러그인
  - MODE_EXPLICIT
    - 향후의 출력 기능을 사용 가능하게 하기 위해서 `setTilingMode` 또는 `setCompressionMode`와 같은 메서드에 건네줄 수 있는 정수값
  - dispose()
    - 이미지 해제 메서드
- IIOImage - 썸네일, 이미지의 메타데이터를 보관 & 유지
- ObjectMetadata - 유저 제공 메타데이터 + HTTP 헤더 포함

```kotlin
fun rotateCompressImage(orientation : Int, width : Double, height : Double, file : File) : File {  
    val atf = AffineTransform()  
    var width = width  
    var height = height  
  
    when (orientation) {  
        1 -> {}  
        2 -> {  
            atf.scale(-1.0, 1.0)  
            atf.translate(-width, 0.0)  
        }  
        3 -> {  
            atf.translate(width, height)  
            atf.rotate(Math.PI)  
        }  
        4 -> {  
            atf.scale(1.0, -1.0)  
            atf.translate(0.0, -height)  
        }  
        5 -> {  
            atf.rotate(-Math.PI / 2)  
            atf.scale(-1.0, 1.0)  
        }  
        6 -> {  
            atf.translate(height, 0.0)  
            atf.rotate(Math.PI / 2)  
        }  
        7 -> {  
            atf.scale(-1.0, 1.0)  
            atf.translate(-height, 0.0)  
            atf.translate(0.0, width)  
            atf.rotate(3 * Math.PI / 2)  
        }  
        8 -> {  
            atf.translate(0.0, width)  
            atf.rotate(3 * Math.PI / 2)  
        }  
    }  
  
    when (orientation) {  
        5, 6, 7, 8 -> {  
            var tempWidth = width  
            width = height  
            height = tempWidth  
        }  
    }  
  
    val image: BufferedImage = ImageIO.read(file)  
    val afterImage = BufferedImage(width.toInt(), height.toInt(), image.type)  
    val rotateOp = AffineTransformOp(atf, AffineTransformOp.TYPE_BILINEAR)  
    val rotatedImage = rotateOp.filter(image, afterImage)  
    val iter = ImageIO.getImageWritersByFormatName("jpg")  
    val writer = iter.next()  
    val iwp = writer.defaultWriteParam  
    iwp.compressionMode = ImageWriteParam.MODE_EXPLICIT  
    iwp.compressionQuality = CalculateImageCompressionRatio(file.length())  
  
    val result = File.createTempFile("compress_image.jpg", "")  
    val fios = FileImageOutputStream(result)  
  
    writer.output = fios  
    writer.write(null, IIOImage(rotatedImage, null, null), iwp)  
    fios.close()  
    writer.dispose()  
  
    return result  
}
```

```kotlin
// 파일로 이미지 업로드
fun uploadImageForFile(files : List<File>, folderName: String) : MutableList<String> {
    
        // 파일 검증
        if (files.isEmpty()) {
            log.error("S3-001 : 업로드하려는 파일이 비었습니다.")
            throw CommonException("S3-001", HttpStatus.BAD_REQUEST)
        }

        // 빈 리스트 생성
        val result: MutableList<String> = mutableListOf()

        // 파일 리스트를 돌면서 파일 이름이 Null 이면 throw Exception
        for (file in files) {
            if (file.name == null) {
                log.error("S3-002 : 이미지 이름이 없어요.")
                throw CommonException("S3-002", HttpStatus.BAD_REQUEST)
            }

            // File -> InputStream 변환
            val fileObj: InputStream = file.inputStream()
            // 파일 생성
            val fileName: String = createFileName(file.name)

            // S3 업로드 (버킷이름, 파일위치, 오브젝트, 메타데이터(파일))
            try {
                ncp.assetNcpS3Client().putObject(
                        PutObjectRequest(
                                bucketName,
                                fileLocation(folderName, fileName),
                                fileObj,
                                imageMetadata(file)
                        )
                )
                // 리스트에 이미지 추가
                result.add(fileLocation(folderName, fileName))
            } catch (e: Exception) {
                log.error("S3-004 : S3 업로드 실패 : $e")
                throw CommonException("S3-004", HttpStatus.BAD_REQUEST)
            }
        }

        log.info("이미지 업로드 성공 [ s3 디렉토리 이름 : $folderName ]")

        return result
    }
```

```kotlin
    fun compress(
        file: File?,
        folder: String
    ) {

        val zipFile = File(folder, "압축.zip")
        val buf: ByteArray = ByteArray(4096)

        try {
            val output: ZipOutputStream = ZipOutputStream(FileOutputStream(zipFile))

            output.use {

                val input: FileInputStream = FileInputStream(file)

                input.use {
                    val ze = ZipEntry(file?.name)
                    output.putNextEntry(ze)

                    val len: Int = input.read(buf)
                    while (len > 0) {
                        output.write(buf, 0, len)
                    }
                    output.closeEntry()
                }

            }
        } catch (_: Exception) {
            log.error("")
        }
        print("압축 파일 생성 성공")
    }

    fun convert(file: MultipartFile): File? {
        val convertFile = File(file.originalFilename)

        if (convertFile.createNewFile()) {
            val fos = FileOutputStream(convertFile)
            fos.use {
                fos.write(file.bytes)
            }
            return convertFile
        }
        return null;
    }
```

```kotlin
class CompressFile(
    private val folder: String = "/img"
) {

    fun compress(
        files: List<File>
    ) {

        val zipFile = File(folder, "압축.zip")
        val buf: ByteArray = ByteArray(4096)

        try {
            val output: ZipOutputStream = ZipOutputStream(FileOutputStream(zipFile))

            output.use {
                for (file in files) {
                    val input: FileInputStream = FileInputStream(file)

                    input.use {
                        val ze = ZipEntry(file.name)
                        output.putNextEntry(ze)

                        val len: Int = input.read(buf)
                        while (len > 0) {
                            output.write(buf, 0, len)
                        }
                        output.closeEntry()
                    }
                }
            }
        } catch (_: Exception) {
        }
        print("압축 파일 생성 성공")
    }
```

```kotlin
    lateinit var regionName: String
    lateinit var accessKey: String
    lateinit var secretKey: String

@Configuration
class NcpS3Config(@Autowired private val ncpProperties: NcpProperties) {

    @Bean
    fun assetNcpS3Client(): AmazonS3 {
        return AmazonS3ClientBuilder.standard()
            .withEndpointConfiguration(
                AwsClientBuilder.EndpointConfiguration(
                    ncpProperties.endpoint,
                    ncpProperties.regionName
                )
            )
            .withCredentials(
                AWSStaticCredentialsProvider(
                    BasicAWSCredentials(
                        ncpProperties.accessKey,
                        ncpProperties.secretKey
                    )
                )
            )
            .build()
    }
}
```

```java
@Configuration
public class S3Config {
 
    @Value("${cloud.aws.credentials.access-key}")
    public String accessKey;
    @Value("${cloud.aws.credentials.secret-key}")
    public String secretKey;
    @Value("${cloud.aws.region.static}")
    public String region;
 
    @Bean
    @Primary
    public BasicAWSCredentials awsCredentialsProvider(){
        BasicAWSCredentials basicAWSCredentials = new BasicAWSCredentials(accessKey, secretKey);
        return basicAWSCredentials;
    }
 
    @Bean
    public AmazonS3 amazonS3() {
        AmazonS3 s3Builder = AmazonS3ClientBuilder.standard()
                .withRegion(region)
                .withCredentials(new AWSStaticCredentialsProvider(awsCredentialsProvider()))
                .build();
        return s3Builder;
    }
 
    // 추가
    @Bean
    public TransferManager transferManager() {
        TransferManager transferManager = TransferManagerBuilder.standard()
                .withS3Client(amazonS3())
                .build();
        return transferManager;
    }
}
```
